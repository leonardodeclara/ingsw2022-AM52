683     /**
684      * Method calculateWinner checks and, if present, sets the winner name by counting each player's number of towers and teacher tiles. If a tie is reached winner is set to null.
685      */
686     public void calculateWinner(){
687         int minTowers = Constants.MAX_TOWER_NUMBER;
688         ArrayList<Player> potentialWinners = new ArrayList<>();
689         for (Player potentialWinner: players){
690             if (potentialWinner.getBoard().getTowers()<minTowers){
691                 potentialWinners.clear();
692                 potentialWinners.add(potentialWinner);
693                 minTowers=potentialWinner.getBoard().getTowers();
694             }
695             else if (potentialWinner.getBoard().getTowers()==minTowers)
696                 potentialWinners.add(potentialWinner);
697         }
698         if (potentialWinners.size() == 1)
699             winner = potentialWinners.get(0);
700         else {
701             int maxTeachers = -1;
702             for (Player finalPlayer : potentialWinners) {
703                 if (finalPlayer.getBoard().getTeacherTable().size()>maxTeachers){
704                     winner = finalPlayer;
705                     maxTeachers= finalPlayer.getBoard().getTeacherTable().size();
706                 }
707                 else if (finalPlayer.getBoard().getTeacherTable().size()==maxTeachers)
708                     winner = null;
709             }
710         }
711     }
