72      * Method handleMessageFromServer receives a Message instance from ClientSocket and react accordingly to the message type.
73      * @param message received from server.
74      * If the message is instance of ClientStateMessage, handleMessageFromServer changes the current client state and if the current scene is updatable, it gets updated
75      * If the message is instance of ErrorMessage, handleMessageFromServer calls the current scene's controller error handling method
76      * If the message is instance of UpdateMessage, handleMessageFromServer updates the GameBoard instance and updates the scene if it's in WAIT_TURN state
77      * Since this method is called from another thread, Platform.runLater() ensures that the method is executed by this (GUI) thread to avoid run time exceptions
78      */
79     public void handleMessageFromServer(Message message){
80         Platform.runLater(new Runnable() {
81             @Override
82             public void run() {
83                 if(message instanceof ClientStateMessage){
84                     currentState = ((ClientStateMessage) message).getNewState();
85                     renderScene();
86                     GUIController currentController = controllers.get(stage.getScene());
87                     if(currentController instanceof UpdatableController){
88                         ((UpdatableController) currentController).update();
89                     }
90                 }
91                 else if (message instanceof ErrorMessage){
92                     GUIController currentController = controllers.get(stage.getScene());
93                     currentController.handleErrorMessage(true);
94                 }
95                 else {
96                     ((UpdateMessage) message).update(GB);
97                     GUIController currentController = controllers.get(stage.getScene());
98                     if(currentController instanceof UpdatableController && currentState.equals(ClientState.WAIT_TURN)){
99                         if(message instanceof ActivePersonalityMessage
100                                 || message instanceof CloudUpdateMessage
101                                 || message instanceof IslandMergeUpdateMessage
102                                 || message instanceof IslandStudentsUpdateMessage
103                                 || message instanceof IslandTowersUpdateMessage
104                                 || message instanceof MotherNatureMovementUpdateMessage)
105                         ((UpdatableController) currentController).update();
106                     }
107                 }
108             }
109         });
110 
111     }

144                 setScene(Constants.WIZARD_CHOICE_FXML);
